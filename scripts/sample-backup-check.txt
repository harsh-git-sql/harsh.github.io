-- Check last backup time for all databases on the instance
SELECT
    d.name AS database_name,
    MAX(b.backup_finish_date) AS last_backup_time,
    DATEDIFF(HOUR, MAX(b.backup_finish_date), GETDATE()) AS hours_since_backup
FROM sys.databases d
LEFT JOIN msdb.dbo.backupset b
    ON d.name = b.database_name AND b.type = 'D'
GROUP BY d.name
ORDER BY hours_since_backup DESC;
